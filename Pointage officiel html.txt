function escapeHTML(str) {
    if (!str) return '';
    return String(str).replace(/[&<>'"]/g, match => ({
        '&': '&amp;', '<': '&lt;', '>': '&gt;', "'": '&#39;', '"': '&quot;'
    }[match]));
}

function getISOWeekNumber(dateStr) {
    if(!dateStr) return '';
    const [d, m, y] = dateStr.split('/').map(Number);
    const date = new Date(y, m - 1, d);
    date.setHours(0, 0, 0, 0);
    date.setDate(date.getDate() + 3 - (date.getDay() + 6) % 7);
    const week1 = new Date(date.getFullYear(), 0, 4);
    return 1 + Math.round(((date.getTime() - week1.getTime()) / 86400000 - 3 + (week1.getDay() + 6) % 7) / 7);
}

function updateMode(id, field, value) {
    const inv = state.currentDay.interventions.find(i => i.id === id);
    inv[field] = value;
    if(field === 'mission') { inv.hDebut = ''; inv.hAppel = ''; inv.hArrivee = ''; inv.hf_raw = ''; inv.hFin = ''; inv.j1 = false; inv.arrondi = false; }
    const list = document.getElementById('interventions_list');
    list.innerHTML = '';
    state.currentDay.interventions.forEach(i => renderCard(i.id));
}

function onSiteChange(id, sel) {
    const inv = state.currentDay.interventions.find(i => i.id === id);
    const selectedOption = sel.options[sel.selectedIndex];
    inv.site = sel.value;
    inv.client = selectedOption.getAttribute('data-client') || '';
    const headClient = document.getElementById(`cl_head_${id}`);
    if(headClient) headClient.textContent = inv.client || 'Site...';
    checkTerminer();
}

function onFinChange(id, val) {
    const inv = state.currentDay.interventions.find(i => i.id === id);
    inv.hf_raw = val; 
    
    if(!val) {
        inv.hFin = '';
        inv.arrondi = false;
        inv.j1 = false;
        document.getElementById(`st_${id}`).innerHTML = '';
        checkTerminer();
        return;
    }
    
    const [h, m] = val.split(':').map(Number); 
    const total = h * 60 + m;
    const reste = total % 15;
    if(reste !== 0) {
        const rounded = total + (15 - reste);
        inv.hFin = String(Math.floor(rounded/60)%24).padStart(2,'0')+':'+String(rounded%60).padStart(2,'0');
        inv.arrondi = true;
    } else { 
        inv.hFin = val; 
        inv.arrondi = false; 
    }
    const debutRef = inv.mission === 'Astreinte' ? inv.hAppel : inv.hDebut;
    if(debutRef && inv.hFin) {
        const [h1, m1] = debutRef.split(':').map(Number); const [h2, m2] = inv.hFin.split(':').map(Number);
        inv.j1 = (h2*60+m2) < (h1*60+m1);
    }
    const st = document.getElementById(`st_${id}`);
    st.innerHTML = (inv.j1?'<span class="alert-tag j1">⚠ FIN J+1</span> ':'') + (inv.arrondi?`<span class="alert-tag arrondi">⏱ ARRONDI À ${inv.hFin}</span>`:'');
    checkTerminer();
}

function toggleAccordion(id) {
    document.querySelectorAll('.card-body').forEach(b => { if(b.id !== `body_${id}`) b.classList.add('collapsed'); });
    document.getElementById(`body_${id}`).classList.toggle('collapsed');
}

function updateInv(id, f, v) { 
    const inv = state.currentDay.interventions.find(i => i.id === id);
    if(inv) inv[f] = v; 
    checkTerminer(); 
}

function updateTime(id, f, v) { 
    updateInv(id, f, v); 
    const inv = state.currentDay.interventions.find(i => i.id === id);
    if(inv && inv.hf_raw) onFinChange(id, inv.hf_raw); 
}

function triggerMic(id) { document.getElementById(`txt_${id}`).focus(); }

function deleteCard(id) {
    if(state.currentDay.interventions.length > 1) {
        state.currentDay.interventions = state.currentDay.interventions.filter(i => i.id !== id);
        
        const list = document.getElementById('interventions_list');
        list.innerHTML = '';
        state.currentDay.interventions.forEach(inv => renderCard(inv.id));
        
        checkTerminer();
    }
}

function checkTerminer() {
    const ok = state.currentDay.interventions.every(i => i.site && i.hFin);
    const btn = document.getElementById('btn_terminer');
    
    btn.disabled = !ok;
    
    if (!ok) {
        btn.textContent = "SAISIE INCOMPLÈTE";
    } else {
        btn.textContent = "VALIDER LA JOURNÉE";
    }
}

function terminerJournee() {
    const idx = state.semaineDays.findIndex(d => d.jourLabel === state.currentDay.jourLabel);
    if(idx > -1) state.semaineDays[idx] = state.currentDay; else state.semaineDays.push(state.currentDay);
    renderWeek(); goTo(3, 4);
}

async function envoyerSemaine() {
    const btn = document.getElementById('btn_envoyer');
    btn.disabled = true; btn.textContent = "ENVOI EN COURS...";
    const payload = { tech: state.tech, semaine: state.semaine, jours: state.semaineDays };
    
    try {
        const response = await fetch(APPS_SCRIPT_URL, { 
            method: 'POST', 
            headers: {
                'Content-Type': 'text/plain;charset=utf-8'
            },
            body: JSON.stringify(payload) 
        });
        
        if (!response.ok) throw new Error("Erreur serveur Google");
        
        goTo(4, 5);
    } catch(e) { 
        alert("Erreur de connexion. Vérifiez votre réseau ou le lien Apps Script."); 
        btn.disabled = false; 
        btn.textContent = "ENVOYER AU BUREAU"; 
    }
}
